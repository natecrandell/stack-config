# Configure Kubernetes master on Raspberry Pi 4 running Ubuntu Server 22.04 LTS
	# Pre-reqs: Complete the bootstrapping detailed in another file in this dir.
	# Sources:
	 - https://www.armosec.io/blog/setting-up-kubernetes-cluster/
     - https://computingforgeeks.com/install-kubernetes-cluster-ubuntu-jammy/

# Pre-flight checks:
	# Ensure hostname and hostname -f resolve properly:
		hostname && hostname -f
	# Ensure swap is disabled
		sudo swapoff -a
		sudo vim /etc/fstab
			#/swap.img
		sudo mount -a
		free -m

# Enable kernel modules
	# The vxlan kernel module is required for Flannel, but is not currently available by default for Pi.
	sudo apt install -y linux-modules-extra-raspi
	
	sudo modprobe br_netfilter overlay vxlan
	sudo tee /etc/modules-load.d/k8s.conf <<EOF
		overlay
		br_netfilter
		vxlan
		EOF

# Add some settings to sysctl
	sudo tee /etc/sysctl.d/kubernetes.conf<<EOF
		net.bridge.bridge-nf-call-ip6tables = 1
		net.bridge.bridge-nf-call-iptables = 1
		net.ipv4.ip_forward = 1
		EOF

	sudo sysctl -p
	
# Install some preferred tools
	sudo apt install apt-transport-https ca-certificates curl gnupg2 lsb-release jq

# Install kubelet, kubeadm and kubectl
	sudo curl -fsSLo /usr/share/keyrings/kubernetes-archive-keyring.gpg https://packages.cloud.google.com/apt/doc/apt-key.gpg
	# I don't know why the following specifies 'xenial', but it appears to be valid.
	echo "deb [signed-by=/usr/share/keyrings/kubernetes-archive-keyring.gpg] https://apt.kubernetes.io/ kubernetes-xenial main" | sudo tee /etc/apt/sources.list.d/kubernetes.list
	
	# As of 2022-12 the current k8s version is 1.26.0, but this version isn't working for reasons I don't yet understand.
	sudo apt update
	# sudo apt install kubelet kubeadm kubectl
	sudo apt install kubelet=1.25.5-00 kubeadm=1.25.5-00 kubectl=1.25.5-00
	sudo apt-mark hold kubelet kubeadm kubectl
	
	# At this point kubelet always fails. This isn't a problem, as kubeadm commands will configure and run kubelet as needed for init/join commands. Just stop the kubelet service for now.
	sudo systemctl stop kubelet
	

# Install and configure Docker as container runtime
	# Note: run as unprivileged user
	# Install Docker		# Also referenced https://docs.docker.com/engine/install/ubuntu/
		curl -fsSL https://download.docker.com/linux/ubuntu/gpg | sudo gpg --dearmor -o /usr/share/keyrings/docker-archive-keyring.gpg
		echo "deb [arch=$(dpkg --print-architecture) signed-by=/usr/share/keyrings/docker-archive-keyring.gpg] https://download.docker.com/linux/ubuntu $(lsb_release -cs) stable" | sudo tee /etc/apt/sources.list.d/docker.list > /dev/null
		sudo apt update && sudo apt install docker-ce docker-ce-cli containerd.io
		sudo mkdir -p /etc/systemd/system/docker.service.d

	# Update the default Docker daemon
		echo '{"exec-opts":["native.cgroupdriver=systemd"],"log-driver":"json-file","log-opts":{"max-size":"100m"},"storage-driver":"overlay2"}' | jq -r | sudo tee /etc/docker/daemon.json

		sudo systemctl enable docker && sudo systemctl daemon-reload && sudo systemctl restart docker
		sudo systemctl status docker

	# Enable Docker access from non-root user-space
		sudo usermod -aG docker $USER
		# Log out and back in so that your group membership is reevaluated
		docker ps		# Verify

# Install & configure cri-dockerd as docker engine shim
	# Reference
		- main: https://computingforgeeks.com/install-mirantis-cri-dockerd-as-docker-engine-shim-for-kubernetes/
		- also:
			- https://kubernetes.io/docs/setup/production-environment/container-runtimes/#docker
			- https://github.com/Mirantis/cri-dockerd#to-use-with-kubernetes
	
	VER=$(curl -s https://api.github.com/repos/Mirantis/cri-dockerd/releases/latest|grep tag_name | cut -d'"' -f4 | sed 's/v//g') && wget https://github.com/Mirantis/cri-dockerd/releases/download/v${VER}/cri-dockerd-${VER}.$(dpkg --print-architecture).tgz
	tar xvf cri-dockerd-${VER}.$(dpkg --print-architecture).tgz && sudo mv cri-dockerd/cri-dockerd /usr/local/bin/ && cri-dockerd --version && rm -rf cri-dockerd/ cri-dockerd-*
	sudo chown root: /usr/local/bin/cri-dockerd
	sudo wget https://raw.githubusercontent.com/Mirantis/cri-dockerd/master/packaging/systemd/cri-docker.service https://raw.githubusercontent.com/Mirantis/cri-dockerd/master/packaging/systemd/cri-docker.socket && sudo mv cri-docker.socket cri-docker.service /etc/systemd/system/ && sudo sed -i -e 's,/usr/bin/cri-dockerd,/usr/local/bin/cri-dockerd,' /etc/systemd/system/cri-docker.service
	sudo systemctl daemon-reload && sudo systemctl enable cri-docker.service && sudo systemctl enable --now cri-docker.socket
	sudo systemctl status cri-docker.socket

	# Useful troubleshooting commands:
		sudo ctr plugin ls
		sudo crictl -r unix:///run/cri-dockerd.sock info

# Init cluster
	sudo kubeadm config images pull --cri-socket unix:///run/cri-dockerd.sock
	
	sudo kubeadm init --pod-network-cidr=10.244.0.0/16 --cri-socket unix:///run/cri-dockerd.sock --apiserver-advertise-address=10.1.1.104

	# https://kubernetes.io/blog/2022/11/18/upcoming-changes-in-kubernetes-1-26/#cri-api-removal
	# The above seems to have worked with v1.25.5-00. In v1.26 the removal of support for CRI v1alpha2 causes kubelet to fail with:
		[kubelet-check] The HTTP call equal to 'curl -sSL http://localhost:10248/healthz' failed with error: Get "http://localhost:10248/healthz": dial tcp 127.0.0.1:10248: connect: connection refused.
	# Just noticed that cri-tools=1.25.0-00 was installed as a dependency. 
		# found latest: https://github.com/kubernetes-sigs/cri-tools/releases/tag/v1.26.0
		# I think I'm going to give up on k8s 1.26 for now.
