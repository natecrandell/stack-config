nano /etc/network/interfaces
auto wlan0 
iface wlan0 inet dhcp
     wpa-driver wext
     wpa-ssid Your_SSID
     wpa-ap-scan 1
     wpa-proto RSN
     wpa-pairwise CCMP
     wpa-group CCMP
     wpa-key-mgmt WPA-PSK
     wpa-psk "Your_WPA-Key_ASCII"
