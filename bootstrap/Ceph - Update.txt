### Upgrade Ceph 2022-12

# RESOURCES
	- https://download.ceph.com/
	
# ISSUES
	- Ubuntu 22.04 cannot use Ceph's debian-pacific focal repo, as the package dependencies do not align.
	- ceph-deploy fails to install --release pacific on Ubuntu 22.04 host because there is no availability for jammy. (https://download.ceph.com/debian-pacific/dists/)	

# MON/MGR/MDS
	# Set the ceph repo (see below)
	# Restart services
		# Use this command to determine service names:
		systemctl list-units --type=service | awk '{print $2}'

# OSDs
	# Set the ceph repo
		echo "deb https://download.ceph.com/debian-pacific/ focal main" > /etc/apt/sources.list.d/ceph.list && apt-get update
		apt-cache policy ceph		# Verify
		apt-get upgrade

	# Workaround required after upgrade on Odroid boards
		sed -i 's/^MemoryDenyWriteExecute=true/MemoryDenyWriteExecute=false/g' /lib/systemd/system/ceph-osd@.service && systemctl daemon-reload
		systemctl restart ceph-osd@0.service

	# Rollback
		echo "deb https://download.ceph.com/debian-16.2.5/ focal main" > /etc/apt/sources.list.d/ceph.list && apt-get update && apt-cache policy ceph
		apt-get install ceph=16.2.5-1focal ceph-mgr=16.2.5-1focal ceph-mon=16.2.5-1focal ceph-osd=16.2.5-1focal ceph-mds=16.2.5-1focal ceph-base=16.2.5-1focal libcephfs2=16.2.5-1focal ceph-common=16.2.5-1focal ceph-mgr-modules-core=16.2.5-1focal libjaeger=16.2.5-1focal librbd1=16.2.5-1focal python3-cephfs=16.2.5-1focal python3-ceph-argparse=16.2.5-1focal python3-ceph-common=16.2.5-1focal python3-rados=16.2.5-1focal python3-rbd=16.2.5-1focal python3-rgw=16.2.5-1focal librados2=16.2.5-1focal librgw2=16.2.5-1focal libradosstriper1=16.2.5-1focal
		reboot
