#Go here to find latest:

dpkg -i zabbix-release_3.4-1+bionic_all.deb

#Create MySQL database and user
#https://www.zabbix.com/documentation/3.0/manual/appendix/install/db_scripts
mysql -uroot -p
create database zabbix character set utf8 collate utf8_bin;
create user 'zabbix'@'localhost' identified by 'Chair9899';
grant all privileges on zabbix.* to 'zabbix'@'localhost' identified by 'Chair9899';
quit;

cd /usr/share/doc/zabbix-server-mysql
#This will take a while.
zcat create.sql.gz | mysql -uroot -p zabbix

vim /etc/zabbix/zabbix_server.conf
DBHost=localhost
DBName=zabbix
DBUser=zabbix
DBPassword=Chair9899

#Start server processes
service zabbix-server start

vim /etc/zabbix/apache.conf
#Uncomment:
